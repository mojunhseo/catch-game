<div id="game-container">

    <div id="score">Score: 0</div>

    <div id="timer">Time: 30</div>

    <img id="basket" src="https://drive.google.com/thumbnail?authuser=0&sz=w200&id=바구니캐릭터ID" alt="바구니">

    

    <div id="msg">

        <h2 id="final-score">게임 종료!</h2>

        <button onclick="location.reload()" style="padding:10px 20px; cursor:pointer;">다시 하기</button>

    </div>

<script>

    const container = document.getElementById('game-container');

    const basket = document.getElementById('basket');

    const scoreDisplay = document.getElementById('score');

    const timerDisplay = document.getElementById('timer');

    const msg = document.getElementById('msg');

    

    let score = 0;

    let timeLeft = 30;

    let gameActive = true;


    // 떨어질 아이템 이미지 리스트 (마케터님의 사진 ID들로 교체!)

    const items = [

        'https://drive.google.com/thumbnail?authuser=0&sz=w200&id=1wcH5pkmuceItTUJgdo9q7bB_e2OVt5J4',

        'https://drive.google.com/thumbnail?authuser=0&sz=w200&id=1wcH5pkmuceItTUJgdo9q7bB_e2OVt5J4'

    ];


    // 바구니 이동 (마우스/터치)

    window.addEventListener('mousemove', (e) => { if(gameActive) basket.style.left = (e.clientX - 40) + 'px'; });

    window.addEventListener('touchmove', (e) => { if(gameActive) basket.style.left = (e.touches[0].clientX - 40) + 'px'; });


    // 아이템 생성

    function createItem() {

        if (!gameActive) return;

        const item = document.createElement('img');

        item.src = items[Math.floor(Math.random() * items.length)];

        item.classList.add('item');

        item.style.left = Math.random() * (window.innerWidth - 50) + 'px';

        item.style.top = '-50px';

        container.appendChild(item);


        let fallInterval = setInterval(() => {

            if (!gameActive) { clearInterval(fallInterval); item.remove(); return; }

            let top = parseInt(item.style.top);

            item.style.top = (top + 5) + 'px';


            // 충돌 감지 (캐치!)

            const bRect = basket.getBoundingClientRect();

            const iRect = item.getBoundingClientRect();


            if (iRect.bottom >= bRect.top && iRect.left < bRect.right && iRect.right > bRect.left) {

                score++;

                scoreDisplay.innerText = "Score: " + score;

                clearInterval(fallInterval);

                item.remove();

            }


            if (top > window.innerHeight) { clearInterval(fallInterval); item.remove(); }

        }, 20);

    }


    let itemGenerator = setInterval(createItem, 800);


    // 타이머

    let countdown = setInterval(() => {

        timeLeft--;

        timerDisplay.innerText = "Time: " + timeLeft;

        if (timeLeft <= 0) {

            gameActive = false;

            clearInterval(countdown);

            clearInterval(itemGenerator);

            document.getElementById('final-score').innerText = "최종 점수: " + score + "점!";

            msg.style.display = 'block';

        }

    }, 1000);

</script>
